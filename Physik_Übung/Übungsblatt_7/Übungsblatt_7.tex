        &= \left( \text{i}\hbar \hat{p}_y\hat{p}_x - \hat{p}_x\text{i}\hbar\hat{p}_y - \text{i}\hbar \hat{p}_x\hat{p}_y + \hat{p}_y\text{i}\hbar\hat{p}_x \right) \Psi\\
        &= 2\text{i}\hbar \underbrace{\left[ \hat{p}_y,\hat p _x \right]}_{=0} \Psi = 0\\
    \end{align*}
